[docs]def video_feed():
    def gen():
        while True:
            gevent.sleep(0.033)
            frame = camera.get_frame()
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + frame + b'\r\n')
    # Return JPEG stream to client video feed.
    return Response(stream_with_context(gen()),mimetype='multipart/x-mixed-replace; boundary=frame')
